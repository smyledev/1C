## Решение

### Задание 1

```bsl
&НаКлиенте
Процедура ПриОткрытии(Отказ)
	ПолеВвода = 8;
КонецПроцедуры

&НаКлиенте
Процедура ЗаполнитьСписок(Команда)
	
	СписокЧисел.Очистить();
	
	ВведенноеЗначение = Число(Элементы.ВведенноеЗначение.ТекстРедактирования);
	Генератор = Новый ГенераторСлучайныхЧисел();      		
	
	СлучайноеЧисло = 0;
	Для Счетчик = 1 По ВведенноеЗначение Цикл 
		СлучайноеЧисло = 0;
		Пока СлучайноеЧисло = 0 Цикл
			СлучайноеЧисло = Генератор.СлучайноеЧисло(1, ВведенноеЗначение * ВведенноеЗначение);
			Если Генератор.СлучайноеЧисло(1, 2) = 1 Тогда
				СлучайноеЧисло = -1 * СлучайноеЧисло;				
			КонецЕсли;			
		КонецЦикла;				
		СтрокаТЧ = СписокЧисел.Добавить();
		СтрокаТЧ.Число = СлучайноеЧисло;
	КонецЦикла;  
	
	ВыделитьМаксимальноеИМинимальноеЧисло();
	
КонецПроцедуры   

Процедура ВыделитьМаксимальноеИМинимальноеЧисло()  
	Максимальное = -256;
	Минимальное = 256;	
	Для Каждого ЧислоИзСписка Из СписокЧисел Цикл
		ТекЧисло = ЧислоИзСписка.Число;
		Если Максимальное < ТекЧисло Тогда
			Максимальное = ТекЧисло;	
		КонецЕсли;		
		Если Минимальное > ТекЧисло Тогда
			Минимальное = ТекЧисло;	
		КонецЕсли;    
	КонецЦикла;		
	МинимальноеЧисло = Минимальное;
	МаксимальноеЧисло = Максимальное;		
КонецПроцедуры

&НаКлиенте
Процедура УдалитьОтрицательные(Команда) 
	КоличествоУдаленныхЧисел = 0; 
	КоличествоЭлементов = СписокЧисел.Количество();
	Счетчик = 0; 
	ТекЧисло = 0;
	МассивУдаленных = Новый Массив;
	Пока Счетчик < КоличествоЭлементов Цикл
		ТекЧисло = СписокЧисел[Счетчик].Число;
		Если ТекЧисло < 0 Тогда
			МассивУдаленных.Добавить(ТекЧисло);
			СписокЧисел.Удалить(Счетчик);	
			КоличествоЭлементов = КоличествоЭлементов - 1;
		Иначе
			Счетчик = Счетчик + 1;			
		КонецЕсли;				
	КонецЦикла;
	
	СписокЗначений = Новый СписокЗначений;
	СписокЗначений.ЗагрузитьЗначения(МассивУдаленных);
	СписокЗначений.СортироватьПоЗначению(НаправлениеСортировки.Убыв);
	МассивУдаленных = СписокЗначений.ВыгрузитьЗначения(); 
	
	ТекстСообщения = "";	
	Для Каждого Элемент Из МассивУдаленных Цикл
		ТекстСообщения = ТекстСообщения + Строка(Элемент) + ", ";	
	КонецЦикла; 	
	ТекстСообщения =  Сред(ТекстСообщения, 0, СтрДлина(ТекстСообщения) - 2);	
	ТекстСообщения = "Удалено значений: " + Строка(МассивУдаленных.Количество()) + " (" + ТекстСообщения + " )";                              
	Сообщить(ТекстСообщения);
	
	ВыделитьМаксимальноеИМинимальноеЧисло();
	
КонецПроцедуры

```
