# Коллекции


## Массив
```
МассивСотрудников = Новый Массив;
  
МассивСотрудников.Добавить("Иванов");
МассивСотрудников.Добавить("Петров");

// Обращение к значениям массива
Иванов = МассивСотрудников[0]; // Иванов
Петров = МассивСотрудников.Получить(1); // Иванов

МассивСотрудников.Удалить(1);

КоличествоСотрудников = МассивСотрудников.Количество();

МассивСотрудников.Добавить("Сидоров");
МассивСотрудников.Добавить("Александрова");

Для каждого ТекСотрудник Из МассивСотрудников Цикл  
  Сообщить(ТекСотрудник); 
КонецЦикла; 


// Присвоить элементу с индексом 1 значение "Вт"...
ДниНедели.Установить(1, "Вт");

// Получить индекс элемента с значением "Вторник"...
ВторойДеньНедели = ДниНедели.Найти("Вторник");

// Получить значение по индексу.
// Вариант 1
ВторойДеньНедели = ДниНедели.Получить(1);

// Вариант 2 
ВторойДеньНедели = ДниНедели[1];
```

## Структура

> Ключ в структуре всегда строка, в массиве - число (индекс)

```
ДанныеСотрудника = Новый Структура;
  
// Добавление значений в структуру
ДанныеСотрудника.Вставить("ФИО", "Иванов");
ДанныеСотрудника.Вставить("Оклад", 1200);
ДанныеСотрудника.Вставить("ДатаРождения", '19910211');

Сообщить(ДанныеСотрудника["ФИО"]);
Сообщить(ДанныеСотрудника["ДатаРождения"]);

ДанныеСотрудника.ФИО = "Иванов Иван Сергеевич";

ДанныеСотрудника.Удалить("Оклад");

Если ДанныеСотрудника.Свойство("КоличествоДетей") = Ложь Тогда
  Сообщить("В структуре данных нет свойства КоличествоДетей");
КонецЕсли; 

Количество = ДанныеСотрудника.Количество();

Для каждого ТекЭлемент Из ДанныеСотрудника Цикл
  Сообщить(ТекЭлемент.Значение);
КонецЦикла; 


// Конструктор для структуры с передачей параметров...
Урок = Новый Структура("Название, НомерКабинета, Начало", "Математика", 311, Дата(2015, 9, 1, 9, 0, 0));


// Хотим получить свойство ФамилияПреподавателя, если его нет, тогда выводим сообщение.
// Возврат позволит остановить продолжение работы программы.
Преподаватель = Неопределено;

Если НЕ Урок.Свойство("ФамилияПреподавателя", Преподаватель) Тогда
  ПоказатьПредупреждение(, "Неизвестно кто ведет урок");
  Возврат;
КонецЕсли;
```


## Соотвествие

> Ключ в соотвествии произвольный
```
ПроизвольныеДанные = Новый Соответствие;

ПроизвольныеДанные.Вставить("ФИО", "Иванов");
ПроизвольныеДанные.Вставить('19910101', "Выходной");
ПроизвольныеДанные.Вставить(44, "XL");

ТипДня = ПроизвольныеДанные['19910101'];
Размер = ПроизвольныеДанные.Получить(44);

Для Каждого ТекЭлемент Из ПроизвольныеДанные Цикл
  Сообщить("" + ТекЭлемент.Ключ + " - " + ТекЭлемент.Значение); 
КонецЦикла;
```
